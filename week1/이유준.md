3.
```c
#include <stdio.h>

int main(){
    int num;
    char str1[] = "number";
    char str2[] = "decimal";
    char str3[] = "octal";
    char str4[] = "hexadecimal";

    printf("양의 정수를 입력하시오 : ");
    scanf("%d", &num);

    printf("|%-12s|%-12s|%-12s|%-12s\n", str1, str2, str3, str4);
    printf("|%-12d|+%-11d|%-12o|%-12X", num, num, num, num);

    return 0;
}
```
